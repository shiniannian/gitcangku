#include <stdio.h>
int main()
{
    char zifu;
    scanf("%c", &zifu);
    if (zifu >= 'A' && zifu <= 'Z')
        printf("It is an alphabetic character.");
    else if (zifu >= 'a' && zifu <= 'z')
        printf("It is an alphabetic character.");
    else if (zifu >= '0' && zifu <= '9')
        printf("It is a digit.");
    else
        printf("It is other character.");

    return 0;
}
